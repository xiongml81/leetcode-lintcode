class Solution {
    public List<List<Integer>> permuteUnique(int[] nums) {
           List<List<Integer>> res = new ArrayList<>();
        boolean[] visited = new boolean[nums.length];
        Arrays.sort(nums);
        dfs(res, visited, nums, new ArrayList<Integer>());
        //remove duplicate
        Set<List<Integer>> set = new HashSet<>();
        for(int i = 0; i < res.size(); i++){
            set.add(res.get(i));
        }
        return new ArrayList<>(set);
    }
    
    public void dfs(List<List<Integer>> res, boolean[] visited, int[] nums, List<Integer> subset){
        //base case
        if(subset.size() == nums.length){
            res.add(new ArrayList<Integer>(subset));
            return;
        }
        
        //拆解
        for(int i = 0; i < nums.length; i++){
            if(visited[i]) continue;
            
            int value = nums[i];
            subset.add(value);
            visited[i] = true;
            dfs(res, visited, nums, subset);
            subset.remove(subset.size() - 1);
            visited[i] = false;
        }
    }
}

class Solution {
    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        boolean[] visited = new boolean[nums.length];
        Arrays.sort(nums);
        dfs(res, visited, nums, new ArrayList<Integer>());
       
        return res;
    }
    
    public void dfs(List<List<Integer>> res, boolean[] visited, int[] nums, List<Integer> subset){
        //base case
        if(subset.size() == nums.length){
            res.add(new ArrayList<Integer>(subset));
            return;
        }
        
        //拆解
        for(int i = 0; i < nums.length; i++){
            if(i >= 1 && nums[i] == nums[i - 1] && !visited[i-1]) continue;
            else if(visited[i]) continue;
            else{
                 int value = nums[i];
                subset.add(value);
                visited[i] = true;
                dfs(res, visited, nums, subset);
                subset.remove(subset.size() - 1);
                visited[i] = false;
            }
           
        }
    }
}
