# 437. Path Sum III

兵分三路，root + left + right为开始点做dfs

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> cur = new ArrayList<>();
        dfs(res, cur, targetSum, root);
        return res;
    }
    
    public void dfs( List<List<Integer>> res, List<Integer> cur, int targetSum, TreeNode root){
        if(root == null){
            return;
        }
        
        if(root != null && root.left == null && root.right == null){
            if(targetSum == root.val){
                cur.add(root.val);
                res.add(new ArrayList<>(cur));
                cur.remove(cur.size() - 1);
                return;
            }
        }
                                                                    
        cur.add(root.val);
        dfs(res, cur, targetSum - root.val,  root.left);
        dfs(res, cur, targetSum - root.val,  root.right);     
        cur.remove(cur.size() - 1);
    }
}
```
