# 94. Binary Tree Inorder Traversal

```javascript
class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        Stack<TreeNode> stack = new Stack<>();

        List<Integer> result = new ArrayList<>();
        TreeNode curr = root;
        if (root == null) {
            return result;
        }

        while(!stack.isEmpty() || curr != null){
            while(curr != null){
                stack.push(curr);
                curr = curr.left;
            }
            curr = stack.pop();
            result.add(curr.val);
            curr = curr.right;
        }
        return result;
    }  
}
```
