# 474. Ones and Zeroes

DFS

```java
class Solution {
    public int findMaxForm(String[] strs, int m, int n) {
        return dfs(strs, m, n, 0);
    }
    
    public int dfs(String[] strs, int m, int n, int index){
        if(index == strs.length){
            return 0;
        }
        int[] t = count(strs[index]);
        int countZero = t[0];
        int countOne =  t[1];
        
        int a = 0;
        if(m - countZero >= 0 && n - countOne >= 0){
            a = 1 + dfs(strs, m - countZero, n - countOne, index + 1);
        }
        int b = dfs(strs, m, n, index + 1);
        return Math.max(a, b);
    }
    
    
    public int[] count(String s){
        int[] res = new int[2];
        for(char c : s.toCharArray()){
            res[c - '0']++;
        }
        return res;
    }
}
```

Memo

```java
class Solution {
    Integer[][][] memo;
    public int findMaxForm(String[] strs, int m, int n) {
        memo = new Integer[strs.length][m + 1][n + 1];
        return dfs(strs, m, n, 0);
    }
    
    public int dfs(String[] strs, int m, int n, int index){
        if(index == strs.length){
            return 0;
        }
        
        if(memo[index][m][n] != null){
            return memo[index][m][n];
        }
        
        int[] t = count(strs[index]);
        int countZero = t[0];
        int countOne =  t[1];
        
        int a = 0;
        if(m - countZero >= 0 && n - countOne >= 0){
            a = 1 + dfs(strs, m - countZero, n - countOne, index + 1);
        }
        int b = dfs(strs, m, n, index + 1);
        memo[index][m][n] = Math.max(a, b);
        return Math.max(a, b);
    }
    
    
    public int[] count(String s){
        int[] res = new int[2];
        for(char c : s.toCharArray()){
            res[c - '0']++;
        }
        return res;
    }
}
```
