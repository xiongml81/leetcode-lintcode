# 300. Longest Increasing Subsequence

```java
class Solution {
    public int lengthOfLIS(int[] nums) {
        int m = nums.length;
        
        int[] array = new int[m];
        
        for(int i = 0; i < m; i++){
            array[i] = 1;
        }
        //array 用来记录数字递增情况，
        for(int i = 1; i < m; i++){
            for(int j = 0; j < i; j++){//里面循环看是否递增次数有变化
                if(nums[i] > nums[j]){
                    array[i] = Math.max(array[i], array[j] + 1);
                }
            }
        }
        //find the max value from array
        int max = 1;
        for(int i = 0; i < m; i++){
            max = Math.max(max, array[i]);
        }
        return max;
    }
}j
```
